    function calcula_numero_dia_semana($dia,$mes,$ano){
    $numerodiasemana = date('w', mktime(0,0,0,$mes,$dia,$ano));
    if ($numerodiasemana == 0) 
        $numerodiasemana = 6;
    else
        $numerodiasemana--;
    return $numerodiasemana;
}
