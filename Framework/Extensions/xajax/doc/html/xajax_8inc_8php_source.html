00484         {
00485 //SkipDebug
00486                 // Check to see if headers have already been sent out, in which case we can't do our job
00487                 if (headers_sent($filename, $linenumber)) {

00496                         // Use xajax error handler if necessary

00498                                 $GLOBALS['xajaxErrorHandlerText'] = "";
00499                                 set_error_handler("xajaxErrorHandler");

00503 
00504                         // handle beforeProcessing event

00824                 Function: _detectURI
00825 
00826                 Returns the current requests URL based upon the SERVER vars.
00827 
00828                 Returns:
00829 
00830                 string : The URL of the current request.
00831         */
00832         function _detectURI() {
00833                 $aURL = array();
00834 
00835                 // Try to get the request URL
00836                 if (!empty($_SERVER['REQUEST_URI'])) {
00837                 
00838                         $_SERVER['REQUEST_URI'] = str_replace(
00839                                 array('"',"'",'<','>'), 
00840                                 array('%22','%27','%3C','%3E'), 
00841                                 $_SERVER['REQUEST_URI']
00842                                 );
00843                                 
00844                         $aURL = parse_url($_SERVER['REQUEST_URI']);
00845                 }
00846 
00847                 // Fill in the empty values
00848                 if (empty($aURL['scheme'])) {
00849                         if (!empty($_SERVER['HTTP_SCHEME'])) {
00850                                 $aURL['scheme'] = $_SERVER['HTTP_SCHEME'];
00851                         } else {
00852                                 $aURL['scheme'] = 
00853                                         (!empty($_SERVER['HTTPS']) && strtolower($_SERVER['HTTPS']) != 'off') 
00854                                         ? 'https' 
00855                                         : 'http';
00856                         }
00857                 }
00858 
00859                 if (empty($aURL['host'])) {
00860                         if (!empty($_SERVER['HTTP_X_FORWARDED_HOST'])) {
00861                                 if (strpos($_SERVER['HTTP_X_FORWARDED_HOST'], ':') > 0) {
00862                                         list($aURL['host'], $aURL['port']) = explode(':', $_SERVER['HTTP_X_FORWARDED_HOST']);
00863                                 } else {
00864                                         $aURL['host'] = $_SERVER['HTTP_X_FORWARDED_HOST'];
00865                                 }
00866                         } else if (!empty($_SERVER['HTTP_HOST'])) {
00867                                 if (strpos($_SERVER['HTTP_HOST'], ':') > 0) {
00868                                         list($aURL['host'], $aURL['port']) = explode(':', $_SERVER['HTTP_HOST']);
00869                                 } else {
00870                                         $aURL['host'] = $_SERVER['HTTP_HOST'];
00871                                 }
00872                         } else if (!empty($_SERVER['SERVER_NAME'])) {
00873                                 $aURL['host'] = $_SERVER['SERVER_NAME'];
00874                         } else {
00875                                 echo $this->objLanguageManager->getText('DTCTURI:01');
00876                                 echo $this->objLanguageManager->getText('DTCTURI:02');
00877                                 exit();
00878                         }
00879                 }
00880 
00881                 if (empty($aURL['port']) && !empty($_SERVER['SERVER_PORT'])) {
00882                         $aURL['port'] = $_SERVER['SERVER_PORT'];
00883                 }
00884 
00885                 if (!empty($aURL['path']))
00886                         if (0 == strlen(basename($aURL['path'])))
00887                                 unset($aURL['path']);
00888                 
00889                 if (empty($aURL['path'])) {
00890                         $sPath = array();
00891                         if (!empty($_SERVER['PATH_INFO'])) {
00892                                 $sPath = parse_url($_SERVER['PATH_INFO']);
00893                         } else {
00894                                 $sPath = parse_url($_SERVER['PHP_SELF']);
00895                         }
00896                         if (isset($sPath['path']))
00897                                 $aURL['path'] = str_replace(array('"',"'",'<','>'), array('%22','%27','%3C','%3E'), $sPath['path']);
00898                         unset($sPath);
00899                 }
00900 
00901                 if (empty($aURL['query']) && !empty($_SERVER['QUERY_STRING'])) {
00902                         $aURL['query'] = $_SERVER['QUERY_STRING'];
00903                 }
00904 
00905                 if (!empty($aURL['query'])) {
00906                         $aURL['query'] = '?'.$aURL['query'];
00907                 }
00908 
00909                 // Build the URL: Start with scheme, user and pass
00910                 $sURL = $aURL['scheme'].'://';
00911                 if (!empty($aURL['user'])) {
00912                         $sURL.= $aURL['user'];
00913                         if (!empty($aURL['pass'])) {
00914                                 $sURL.= ':'.$aURL['pass'];
00915                         }
00916                         $sURL.= '@';
00917                 }
00918 
00919                 // Add the host
00920                 $sURL.= $aURL['host'];
00921 
00922                 // Add the port if needed
00923                 if (!empty($aURL['port']) 
00924                         && (($aURL['scheme'] == 'http' && $aURL['port'] != 80) 
00925                         || ($aURL['scheme'] == 'https' && $aURL['port'] != 443))) {
00926                         $sURL.= ':'.$aURL['port'];
00927                 }
00928 
00929                 // Add the path and the query string
00930                 $sURL.= $aURL['path'].@$aURL['query'];
00931 
00932                 // Clean up
00933                 unset($aURL);
00934                 
00935                 $aURL = explode("?", $sURL);
00936                 
00937                 if (1 < count($aURL))
00938                 {
00939                         $aQueries = explode("&", $aURL[1]);
00940 
00941                         foreach ($aQueries as $sKey => $sQuery)
00942                         {
00943                                 if ("xjxGenerate" == substr($sQuery, 0, 11))
00944                                         unset($aQueries[$sKey]);
00945                         }
00946                         
00947                         $sQueries = implode("&", $aQueries);
00948                         
00949                         $aURL[1] = $sQueries;
00950                         
00951                         $sURL = implode("?", $aURL);
00952                 }
00953 
00954                 return $sURL;
00955         }
