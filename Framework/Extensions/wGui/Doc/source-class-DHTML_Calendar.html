100:     function _make_js_hash($array) {
101:         $jstr = '';
102:         reset($array);
103:         while (list($key, $val) = each($array)) {
104:             if (is_bool($val))
105:                 $val = $val ? 'true' : 'false';
106:             else if (!is_numeric($val))
107:                 $val = '"'.$val.'"';
108:             if ($jstr) $jstr .= ',';
109:             $jstr .= '"' . $key . '":' . $val;
110:         }
111:         return $jstr;
112:     }
