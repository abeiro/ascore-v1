    function as_libdate_fecha($stamp="") {
    if (empty($stamp))
        $stamp=time();
    $dia=ucfirst(strftime("%A",$stamp));
    $dia_n=strftime("%d",$stamp)+0;
    $mes=ucfirst(strftime("%B",$stamp));
    $anio=strftime("%Y",$stamp);
    return "$dia, $dia_n de $mes de $anio";
}
