        function del($d) {



            if (!isset($d["num"])||($d["num"]==0))
            $this->contenido["P".$d["ID"]]=-1;
        else
            $this->contenido["P".$d["ID"]]-=$d["num"];
        if ($this->contenido["P".$d["ID"]]<=0) {
            asort($this->contenido);
            array_shift($this->contenido);
        }

        $_SESSION["cartpyme"]=$this->contenido;

        return true;

    }
